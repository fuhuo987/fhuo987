async function http(obj){

//先将形参obj解构赋值

let {method,url,params,data}=obj

//params需要处理，转化成能拼接在url后面的格式

//data需要处理，如果有需要写完整的代码headers

url=url+'?'+str;

}

if(data){

res=await fetch(url,{

method:method,

headers:{

'Content-Type':'application/json'

},

body:JSON.stringify(data)

})

}else{

res=await fetch(url)

}

return res.json();

}
